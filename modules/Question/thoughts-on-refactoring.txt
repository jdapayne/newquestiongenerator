class GraphicQ {
  constructor(options) {
    this.data = new GraphicQData(options)
    this.view = new GraphicQView(this.data, width, height)
    // no reference to this.DOM
  }

  getDOM () {
    return this.view.getDOM()
  }

  render () {
    this.view.render()
  }

  showAnswer() {
    this.view.showAnswer()
  }
